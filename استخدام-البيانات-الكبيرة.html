مفاهيم عامة حول البيانات الكبيرة ســتخدام ســيؤدي إلــى فتــح آفــاق واســعة للبحــث والتطويــر علــى مختلــف الأصعــدة. اإتاحــة البيانــات الكبيــرة بأنواعهــا المختلفــة ل أن ممــا لا شــك فيــه بــأن البيانــات الكبيــرة تتيــح Oracle أن البيانــات الكبيــرة تعطيــك فرصــة لاكتشــاف رؤى مهمــة فــي البيانــات، وتضيــف شــركة اوراكل IBMتقــول شــركة للشــركات أن تفهــم زبائنهــا بعمــق أكثــر. إن الفائــدة المتوقعــة مــن اســتخدام البيانــات الكبيــرة ســتكون فــي مختلــف الجوانــب والمجــالات، وعلــى ســبيل المثــال لا الحصــر:

 تشــجيع الإبــداع والابتــكار، مــن خــال اســتغلالقواعــد البيانــات الكبيــرة الحاليــة وإجــراء دراســات تفصيليــة عليهــا لتطويــر وابتــكار خدمــات جديــدة تســاهم فــي تحســين مســتوى أداء المؤسســة أو الشــركة.   
  •     دعــم البحــث العلمــي، وذلــك فــي مختلــف الحقــول والمجــالات، مــن خــال توفيرهــا لقواعــد بيانــات علــى مســتويات تفصيليــة للمجتمعــات المبحوثــة، وضمــن النطــاق الزمنــي المحــدد لمرجعيــة البيانــات. ممــا ينعكــس إيجابــا علــى دقــة وكفــاءة نتائــج الأبحــاث والدراســات
